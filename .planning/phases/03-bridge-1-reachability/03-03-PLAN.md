
Implementation:
- Call `httpGet("https://" + domain, { method: "HEAD", timeout })` -- HEAD request, not GET
- If `result.ok` (any response, even 4xx/5xx): HTTPS works → `{ check: { id: "https_available", label: "HTTPS Available", status: "pass", detail: "HTTPS connection successful" }, abort: false }`
- If `!result.ok`:
  - `error.kind === "dns"`: fail + abort=true, abortReason="dns"
  - `error.kind === "connection_refused"`: fail + abort=true, abortReason="connection_refused"
  - `error.kind === "ssl_error"`: fail + abort=true, abortReason="ssl_error"
  - `error.kind === "timeout"`: fail + abort=false (timeout doesn't abort -- already retried once)
  - `error.kind === "bot_protected"`: fail + abort=false (site exists, just blocked)
  - Any other error: fail + abort=false
- Detail includes error.message for failures

    ```typescript
    function calculateScore(checks: Check[]): { score: number; scoreLabel: "pass" | "partial" | "fail" } {
      let points = 0;
      let maxPoints = 0;

      for (const check of checks) {
        // Skip checks excluded from scoring (crawler policy skip)
        if (check.data && (check.data as Record<string, unknown>).policy === "skip") {
          continue;
        }

        maxPoints += 1;
        if (check.status === "pass") points += 1;
        else if (check.status === "partial") points += 0.5;
        // fail and error = 0 points
      }

      const score = maxPoints === 0 ? 0 : Math.round((points / maxPoints) * 100);
      const scoreLabel = score >= 80 ? "pass" : score >= 40 ? "partial" : "fail";
      return { score, scoreLabel };
    }
    ```
